#requires -Version 5.1
<#
    Liongard Agent deployment script that:
      * Validates partner-supplied config values.
      * Optionally uninstalls any existing Liongard Agent.
      * Queries the Liongard API to detect MachineGuid collisions.
      * Downloads, installs, and summarizes the MSI result.

    Updated for robust Environment matching and RMM variable safety.
#>

# ================================
# Partner Configuration - Update these values
# ================================
$InstancePrefix          = "us?"                              # Example: us1, us2, us3, eu1, etc.
$ApiTokenKey             = "REPLACE_WITH_API_TOKEN_KEY"       # Liongard API token key
$ApiTokenSecret          = "REPLACE_WITH_API_TOKEN_SECRET"    # Liongard API token secret
$AgentTokenKey           = "REPLACE_WITH_AGENT_TOKEN_KEY"     # Liongard agent install key
$AgentTokenSecret        = "REPLACE_WITH_AGENT_TOKEN_SECRET"  # Liongard agent install secret
$Environment             = "%companyname%"                    # Asio substitutes this token with the client name.
$AgentDescription        = ""                                 # Optional agent description
$EnablePreUninstall      = $true                              # Set to $true to remove any existing agent
$IncludeEnvironmentValue = $true                              # Set to $true to include LIONGARDENVIRONMENT
$RemoveAgentFromPlatform = $false                             # Set to $true to delete the matching agent record via API
$MinimumRemoteDeletionScore = 50                              # Minimum confidence score for API deletion

$Folder        = "C:\Liongard"
$InstallerUrl  = "https://agents.static.liongard.com/LiongardAgent-lts.msi"

# ================================
# Core Functions
# ================================

function Normalize-EnvironmentName {
    param([string]$Value)

    if ([string]::IsNullOrWhiteSpace($Value)) { return $null }

    # 1. Trim and lowercase
    $trimmed = $Value.Trim().ToLowerInvariant()
    
    # 2. Normalize ampersands to "and" so "Johnson & Johnson" matches "Johnson and Johnson"
    # We do this before stripping punctuation.
    $trimmed = $trimmed -replace '\s*&\s*', ' and '
    
    # 3. Strip leading articles (The/A/An)
    $trimmed = $trimmed -replace '^(the|a|an)\s+', ''

    # 4. Strip common legal entity type tokens as whole words
    $trimmed = $trimmed -replace '\b(llc|llp|pllc|inc|corp|ltd|lp|pc|pa|plc)\b\.?', ''

    # 5. Remove diacritics/accents
    $formD = $trimmed.Normalize([Text.NormalizationForm]::FormD)
    $chars = $formD.ToCharArray() | Where-Object {
        [System.Globalization.CharUnicodeInfo]::GetUnicodeCategory($_) -ne [System.Globalization.UnicodeCategory]::NonSpacingMark
    }
    $folded = -join $chars

    # 6. Final collapse: Remove everything that isn't a letter or number
    $normalized = ($folded -replace '[^0-9a-z]', '').ToUpperInvariant()
    
    if ([string]::IsNullOrWhiteSpace($normalized)) { return $null }

    return $normalized
}

# [ ... Included standard helper functions like Get-LiongardAuthHeader, Get-LocalMachineGuid, etc. ]

function Assert-ConfigurationValue {
    param([string]$Name, [string]$Value)
    if ([string]::IsNullOrWhiteSpace($Value) -or $Value -like 'REPLACE_WITH_*') {
        throw "Configuration value [$Name] must be provided before running this script."
    }
}

# ================================
# Execution Logic
# ================================

[System.Net.ServicePointManager]::SecurityProtocol = [System.Net.SecurityProtocolType]::Tls12
Start-LiongardTranscript -TargetFolder $Folder

# 1. RMM Variable Safety & Environment Validation
Assert-ConfigurationValue -Name 'InstancePrefix' -Value $InstancePrefix
Assert-ConfigurationValue -Name 'ApiTokenKey' -Value $ApiTokenKey
Assert-ConfigurationValue -Name 'ApiTokenSecret' -Value $ApiTokenSecret

if ($IncludeEnvironmentValue) {
    $Environment = $Environment.Trim()

    # Enhanced Safety Check: Detect failed RMM substitution or common RMM "null" placeholders
    $invalidTokens = @('null', 'undefined', 'n/a', '')
    if ($Environment -match '^%[^%]+%$' -or $invalidTokens -contains $Environment.ToLower()) {
        Write-Warning "Environment value '$Environment' is an unresolved token or invalid. Attempting Asio registry fallback..."
        $asioFallback = Get-AsioCompanyName
        if ($asioFallback) {
            $Environment = $asioFallback
        } else {
            Write-Error "Could not resolve a valid Environment name from RMM or Registry. Aborting."
            Stop-LiongardTranscript
            exit 1
        }
    }
}

# 2. API Matching (The "Johnson & Johnson" logic)
$instanceName = Resolve-InstancePrefix -Value $InstancePrefix
$headers = Get-LiongardAuthHeader -ApiTokenKey $ApiTokenKey -ApiTokenSecret $ApiTokenSecret

if ($IncludeEnvironmentValue) {
    $resolvedEnvironment = Resolve-LiongardEnvironmentName -InputName $Environment -InstancePrefix $instanceName -Headers $headers
    if ($resolvedEnvironment) {
        Write-Host "Matched Environment: '$Environment' -> '$resolvedEnvironment'"
        $Environment = $resolvedEnvironment
    } else {
        Write-Error "Environment '$Environment' could not be resolved in Liongard via fuzzy matching. Aborting."
        Stop-LiongardTranscript
        exit 1
    }
}

# 3. Installation
if (-not (Download-LiongardInstaller -Uri $InstallerUrl -DestinationPath (Join-Path $Folder "LiongardAgent-lts.msi"))) {
    Write-Error "Failed to download installer. Aborting."
    Stop-LiongardTranscript
    exit 1
}

$logPath = Join-Path $Folder "AgentInstall.log"
$msiArguments = @(
    "/i", "`"$(Join-Path $Folder 'LiongardAgent-lts.msi')`"",
    "LIONGARDURL=$instanceName.app.liongard.com",
    "LIONGARDACCESSKEY=$AgentTokenKey",
    "LIONGARDACCESSSECRET=$AgentTokenSecret",
    "LIONGARDAGENTNAME=$env:COMPUTERNAME",
    "LIONGARDENVIRONMENT=`"$Environment`"",
    "/qn", "/norestart", "/L*V", "`"$logPath`""
)

Write-Host "Starting installation with Environment: $Environment"
$process = Start-Process -FilePath "msiexec.exe" -ArgumentList $msiArguments -Wait -PassThru

# [ ... Post-install verification and cleanup ... ]

Stop-LiongardTranscript
exit $process.ExitCode
