#requires -Version 5.1
<#
    Liongard Agent deployment script.
    Optimized for ConnectWise Asio + Special Character Handling.
#>

# ================================
# Partner Configuration
# ================================
$InstancePrefix          = "us?" 
$ApiTokenKey             = "REPLACE_WITH_API_TOKEN_KEY" 
$ApiTokenSecret          = "REPLACE_WITH_API_TOKEN_SECRET"
$AgentTokenKey           = "REPLACE_WITH_AGENT_TOKEN_KEY"
$AgentTokenSecret        = "REPLACE_WITH_AGENT_TOKEN_SECRET"

# Safe ingestion: Uses Here-String to wrap the RMM value safely
$Environment = @'
%companyname%
'@.Trim()

$AgentDescription        = ""                                
$EnablePreUninstall      = $true                             
$IncludeEnvironmentValue = $true                             
$Folder        = "C:\Liongard"
$InstallerUrl  = "https://agents.static.liongard.com/LiongardAgent-lts.msi"

# ================================
# HELPER FUNCTIONS
# ================================

function Normalize-EnvironmentName {
    param([string]$Value)
    if ([string]::IsNullOrWhiteSpace($Value)) { return $null }
    $trimmed = $Value.ToLowerInvariant()
    $trimmed = $trimmed -replace '\s*&\s*', ' and '
    $trimmed = $trimmed -replace '^(the|a|an)\s+', ''
    $trimmed = $trimmed -replace '\b(llc|llp|pllc|inc|corp|ltd|lp|pc|pa|plc)\b\.?', ''
    $formD = $trimmed.Normalize([Text.NormalizationForm]::FormD)
    $chars = $formD.ToCharArray() | Where-Object {
        [System.Globalization.CharUnicodeInfo]::GetUnicodeCategory($_) -ne [System.Globalization.UnicodeCategory]::NonSpacingMark
    }
    return (-join $chars -replace '[^0-9a-z]', '').ToUpperInvariant()
}

function Get-AsioCompanyName {
    $paths = @('HKLM:\SOFTWARE\ITSPlatform', 'HKLM:\SOFTWARE\WOW6432Node\ITSPlatform')
    foreach ($p in $paths) {
        if (Test-Path $p) {
            $val = (Get-ItemProperty $p -ErrorAction SilentlyContinue).CompanyName
            if ($val) { return $val.Trim() }
        }
    }
    return $null
}

function Start-LiongardTranscript {
    param($TargetFolder)
    if (-not (Test-Path $TargetFolder)) { New-Item $TargetFolder -ItemType Directory -Force | Out-Null }
    $script:TranscriptPath = Join-Path $TargetFolder "Liongard_Deploy.log"
    Start-Transcript -Path $script:TranscriptPath -Append -Force | Out-Null
}

function Resolve-LiongardEnvironmentName {
    param($InputName, $Prefix, $Headers)
    $normInput = Normalize-EnvironmentName $InputName
    $uri = "https://$Prefix.app.liongard.com/api/v1/environments"
    try {
        $envs = (Invoke-RestMethod -Uri $uri -Headers $Headers -ErrorAction Stop).data
        # Match 1: Exact string (case insensitive)
        foreach ($e in $envs) { if ($e.Name -ieq $InputName) { return $e.Name } }
        # Match 2: Normalized (Johnson & Johnson -> Johnson and Johnson)
        foreach ($e in $envs) { if (Normalize-EnvironmentName $e.Name -eq $normInput) { return $e.Name } }
    } catch { Write-Warning "API lookup failed: $($_.Exception.Message)" }
    return $null
}

# ================================
# MAIN LOGIC
# ================================
[System.Net.ServicePointManager]::SecurityProtocol = [System.Net.SecurityProtocolType]::Tls12
Start-LiongardTranscript -TargetFolder $Folder

try {
    Write-Host "RMM Provided Environment: [$Environment]"

    # 1. Validation Logic
    if ($Environment -eq '%companyname%' -or [string]::IsNullOrWhiteSpace($Environment) -or $Environment -eq 'null') {
        Write-Warning "RMM substitution failed. Attempting Registry fallback..."
        $Environment = Get-AsioCompanyName
    }

    if ([string]::IsNullOrWhiteSpace($Environment)) {
        throw "No valid Environment name could be resolved. Check Asio Site naming."
    }

    # 2. Scrub quotes to prevent MSI argument break
    # If a name has a ", we swap it to ' to keep the MSI string valid
    $Environment = $Environment -replace '"', "'"

    $cleanPrefix = ($InstancePrefix -split '\.')[0]
    $headers = @{ 
        'X-ROAR-API-KEY' = [System.Convert]::ToBase64String([System.Text.Encoding]::UTF8.GetBytes("$ApiTokenKey:$ApiTokenSecret"))
        'User-Agent' = 'LiongardAgentInstaller/1.0'
    }

    # 3. Fuzzy Match
    $resolvedEnv = Resolve-LiongardEnvironmentName $Environment $cleanPrefix $headers
    if (-not $resolvedEnv) {
        throw "Environment matching failed. Could not find '$Environment' in Liongard."
    }
    $Environment = $resolvedEnv
    Write-Host "Successfully matched to Liongard Environment: $Environment"

    # 4. Uninstall/Install
    if ($EnablePreUninstall) { 
        $svc = Get-Service "LiongardAgent*" -ErrorAction SilentlyContinue
        if ($svc) { Stop-Service $svc.Name -Force; sc.exe delete $svc.Name | Out-Null }
    }

    $msiPath = Join-Path $Folder "LiongardAgent-lts.msi"
    Write-Host "Downloading installer..."
    Invoke-WebRequest -Uri $InstallerUrl -OutFile $msiPath -UseBasicParsing -ErrorAction Stop

    # We use -ArgumentList as an array to handle special characters safely
    $msiArgs = @(
        "/i", "`"$msiPath`"",
        "LIONGARDURL=$cleanPrefix.app.liongard.com",
        "LIONGARDACCESSKEY=$AgentTokenKey",
        "LIONGARDACCESSSECRET=$AgentTokenSecret",
        "LIONGARDENVIRONMENT=`"$Environment`"",
        "LIONGARDAGENTNAME=$env:COMPUTERNAME",
        "/qn", "/norestart"
    )

    Write-Host "Executing MSI..."
    $proc = Start-Process "msiexec.exe" -ArgumentList $msiArgs -Wait -PassThru
    Write-Host "Installation finished. Exit Code: $($proc.ExitCode)"
}
catch {
    Write-Error "CRITICAL FAILURE: $($_.Exception.Message)"
}
finally {
    Stop-Transcript | Out-Null
}
